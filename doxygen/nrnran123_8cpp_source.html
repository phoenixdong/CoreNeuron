33 /** Tag type for use with boost::fast_pool_allocator that forwards to
34  * coreneuron::[de]allocate_unified(). Using a Random123-specific type here
35  * makes sure that allocations do not come from the same global pool as other
36  * usage of boost pools for objects with sizeof == sizeof(nrnran123_State).
37  *
38  * The messy m_block_sizes map is just because `deallocate_unified` uses sized
39  * deallocations, but the Boost pool allocators don't. Because this is hidden
40  * behind the pool mechanism, these methods are not called very often and the
41  * overhead is minimal.
42  */
43 struct random123_allocate_unified {
44  using size_type = std::size_t;
45  using difference_type = std::size_t;
46  static char* malloc(const size_type bytes) {
47  std::lock_guard<std::mutex> const lock{m_mutex};
48  static_cast<void>(lock);
49  auto* buffer = coreneuron::allocate_unified(bytes);
50  m_block_sizes[buffer] = bytes;
51  return reinterpret_cast<char*>(buffer);
52  }
53  static void free(char* const block) {
54  std::lock_guard<std::mutex> const lock{m_mutex};
55  static_cast<void>(lock);
56  auto const iter = m_block_sizes.find(block);
57  assert(iter != m_block_sizes.end());
58  auto const size = iter->second;
59  m_block_sizes.erase(iter);
60  return coreneuron::deallocate_unified(block, size);
61  }
62  static std::mutex m_mutex;
63  static std::unordered_map<void*, std::size_t> m_block_sizes;
64 };

102 void setup_global_state() {
103 #if !defined(CORENEURON_PREFER_OPENMP_OFFLOAD)
104  if (g_k) {
105  // Already initialised, nothing to do
106  return;
107  }
108  g_k = coreneuron::allocate_unique<philox4x32_key_t>(
109  coreneuron::unified_allocator<philox4x32_key_t>{})
110  .release();
111 #ifdef __CUDACC__
112  if (coreneuron::unified_memory_enabled()) {
113  // Set the device-side global g_k_dev to point to the newly-allocated
114  // unified memory. If this is false, g_k is just a host pointer and
115  // there is no point initialising the device global to it.
116  {
117  auto const code = cudaMemcpyToSymbol(g_k_dev, &g_k, sizeof(g_k));
118  static_cast<void>(code);
119  assert(code == cudaSuccess);
120  }
121  // Make sure g_k_dev is updated.
122  {
123  auto const code = cudaDeviceSynchronize();
124  static_cast<void>(code);
125  assert(code == cudaSuccess);
126  }
127  }
128 #endif
129 #endif
130 }

223 /* at cost of a cached value we could compute two at a time. */
224 CORENRN_HOST_DEVICE double nrnran123_normal(nrnran123_State* s) {
225  double w, x, y;
226  double u1, u2;
227 
228  do {
229  u1 = nrnran123_dblpick(s);
230  u2 = nrnran123_dblpick(s);
231  u1 = 2. * u1 - 1.;
232  u2 = 2. * u2 - 1.;
233  w = (u1 * u1) + (u2 * u2);
234  } while (w > 1);
235 
236  y = std::sqrt((-2. * log(w)) / w);
237  x = u1 * y;
238  return x;
239 }

241 CORENRN_HOST_DEVICE double nrnran123_uint2dbl(uint32_t u) {
242  /* 0 to 2^32-1 transforms to double value in open (0,1) interval */
243  /* min 2.3283064e-10 to max (1 - 2.3283064e-10) */
244  return ((double) u + 1.0) * SHIFT32;

264 /** @brief Allocate a new Random123 stream.
265  * @todo It would be nicer if the API return type was
266  * std::unique_ptr<nrnran123_State, ...not specified...>, so we could use a
267  * custom allocator/deleter and avoid the (fragile) need for matching
268  * nrnran123_deletestream calls.
269  */
270 nrnran123_State* nrnran123_newstream3(uint32_t id1,
271  uint32_t id2,
272  uint32_t id3,
273  bool use_unified_memory) {
274  // The `use_unified_memory` argument is an implementation detail to keep the
275  // old behaviour that some Random123 streams that are known to only be used
276  // from the CPU are allocated using new/delete instead of unified memory.
277  // See OPENACC_EXCLUDED_FILES in coreneuron/CMakeLists.txt. If we dropped
278  // this feature then we could always use coreneuron::unified_allocator.
279 #ifndef CORENEURON_ENABLE_GPU
280  if (use_unified_memory) {
281  throw std::runtime_error("Tried to use CUDA unified memory in a non-GPU build.");
282  }
283 #endif
284  nrnran123_State* s{nullptr};
285  if (use_unified_memory) {
286  s = coreneuron::allocate_unique<nrnran123_State>(random123_allocator{}).release();
287  } else {
288  s = new nrnran123_State{};
289  }
290  s->c.v[0] = 0;
291  s->c.v[1] = id3;
292  s->c.v[2] = id1;
293  s->c.v[3] = id2;
294  nrnran123_setseq(s, 0, 0);
295  {
296  // TODO: can I assert something useful about the instance count going
297  // back to zero anywhere? Or that it is zero when some operations happen?
298  std::lock_guard<OMP_Mutex> _{g_instance_count_mutex};
299  ++g_instance_count;
300  }
301  return s;
302 }
